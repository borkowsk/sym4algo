Program Losowanie_galtona2; 
{Jak wyglada rozk³ad wylosowanych cyfr gdy jest to suma wielu losowañ} 

Const 
   ile_razy=10000; 
   ile_losow=7; 
Var 
   liczniki:array[0..ile_losow] of integer; 

procedure losuj; 

Var 
   i,j:integer; 
   c:integer; 
begin 
  WriteLn('Losowanie ',ile_razy,' kulek z ',ile_losow,' przegródkowej deski Galtona'); 
  For i:=1 to ile_razy do 
    begin 
      c:=0; 
      for j:=1 to ile_losow do 
        c:=c+random(2); 
      Write(c,' '); 
      liczniki[c]:=liczniki[c]+1; 
    end; 
end; 

procedure zeruj; 

Var 
   j:integer; 
begin 
  for j:=0 to ile_losow do 
    liczniki[j]:=0; 
end; 

procedure drukuj; 

Var 
   j:integer; 
begin 
  writeln; 
  writeln('WYNIKI:'); 
  for j:=0 to ile_losow do 
    writeln(j,' ',liczniki[j]); 
end; 

procedure zapisz; 

Var 
   j:integer; 
   f:text; 
begin 
  assign(f,'deska.out'); 
  rewrite(f); 
  WriteLn(f,'Losowanie ',ile_razy,' kulek z ',ile_losow,' przegródkowej deski Galtona'); 
  for j:=0 to ile_losow do 
    writeln(f,j,chr(9),liczniki[j]); 
  close(f);
end; 

Begin 
  zeruj; 
  losuj; 
  drukuj; 
  zapisz; 
end.
